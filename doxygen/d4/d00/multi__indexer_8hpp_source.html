248template <typename TType>
249void MultiIndexHandle<TType>::addToGroup(const std::string& group){
250 if (m_alive_flag == nullptr || !m_alive_flag->load()){
251 throw std::runtime_error("The Indexer of this Handle was already destroyed");
252 }
253 auto owner = *m_owner_ptr;
254 auto it = owner->m_groups.find(group);
255 if (it == owner->m_groups.end()){
256 owner->m_groups.try_emplace(group);
257 it = owner->m_groups.find(group);
258 }
259 auto handle = it->second.makeNew(this);
260 m_groups.insert({group, std::move(handle)});
261}
