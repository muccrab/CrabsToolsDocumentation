245template <typename TType>
246void MultiIndexHandle<TType>::addToGroup(const std::string& group){
247 if (m_alive_flag == nullptr || !m_alive_flag->load()){
248 throw std::runtime_error("The Indexer of this Handle was already destroyed");
249 }
250 auto owner = *m_owner_ptr;
251 auto it = owner->m_groups.find(group);
252 if (it == owner->m_groups.end()){
253 owner->m_groups.try_emplace(group);
254 it = owner->m_groups.find(group);
255 }
256 auto handle = it->second.makeNew(this);
257 m_groups.insert({group, std::move(handle)});
258}
