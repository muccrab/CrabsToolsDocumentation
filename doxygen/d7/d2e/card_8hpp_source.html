200 Card& Card::addComponent(){
201 m_components.push_back(std::make_unique<CMP>(this));
202 std::type_index type_idx(typeid(CMP));
203 auto it = m_component_map.find(type_idx);
204 if (it != m_component_map.end()) return *this;
205 m_component_map.insert({type_idx, m_components.size() - 1});
206 return *this;
207 }
208
209 template <typename CMP>
210 const CMP* Card::getComponent() const{
211 std::type_index type_idx(typeid(CMP));
212 auto it = m_component_map.find(type_idx);
213 if (it == m_component_map.end()) return nullptr;
214 return dynamic_cast<const CMP*>(m_components[it->second].get());
215 }
