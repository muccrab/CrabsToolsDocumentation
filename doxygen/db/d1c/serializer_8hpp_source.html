188 SerializerValue ISerializer::getValue(const std::string& name) const {
189 auto accessor_item = m_accessor_map.find(name);
190 if (accessor_item == m_accessor_map.end()){
191 return SerializerValue(SERIALIZER_NULL, ValueTypeEnum::undefined);
192 }
193 auto accessor = m_accessors[accessor_item->second];
194 if (accessor.second.access_cache() != SerializerCacheEnum::value){
195 return SerializerValue(SERIALIZER_NULL, ValueTypeEnum::undefined);
196 }
197 return m_values[accessor.second.index()];
198 }
